    <script>
        function calculateBMI() {
            const hInput = document.getElementById('height');
            const wInput = document.getElementById('weight');
            const resultBox = document.getElementById('result-box');
            const bmiVal = document.getElementById('bmi-val');
            const bmiStatus = document.getElementById('bmi-status');


            const h = hInput.value / 100;
            const w = wInput.value;


            if (!h || !w) {
                alert("Please enter both height and weight.");
                return;
            }


            const bmi = (w / (h * h)).toFixed(1);
            
            resultBox.style.display = "block";
            bmiVal.innerHTML = bmi;
            
            resultBox.className = ""; // Reset classes


            if (bmi < 18.5) {
                bmiStatus.innerHTML = "Category: Underweight";
                resultBox.classList.add('bmi-warning');
            } else if (bmi >= 18.5 && bmi < 24.9) {
                bmiStatus.innerHTML = "Category: Healthy Weight";
                resultBox.classList.add('bmi-normal');
            } else if (bmi >= 25 && bmi < 29.9) {
                bmiStatus.innerHTML = "Category: Overweight";
                resultBox.classList.add('bmi-warning');
            } else {
                bmiStatus.innerHTML = "Category: Obese";
                resultBox.classList.add('bmi-danger');
            }
        }
